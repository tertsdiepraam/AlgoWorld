function insertion_sort(array A)
    i ← 1
    while i < length(A)
        x ← A[i]
        j ← i - 1
        while j >= 0 and A[j] > x
            A[j+1] ← A[j]
            j ← j - 1
        end while
        A[j+1] ← x[4]
        i ← i + 1
    end while

Rust

fn insertion_sort<T>(a: &mut [T]) 
    where T: Ord
{
    for i in 1..a.len() {
        let mut j = i;
        while j > 0 && a[j-1] > a[j] {
            a.swap(j, j-1);
            j -= 1;
        }
    }
}

Python

def insertion_sort(a):
    for i in range(1, len(a)):
        j = i
        while j > 0 and a[j-1] > a[j]:
            a[j-1], a[j] = a[j], a[j-1]
            j -= 1
